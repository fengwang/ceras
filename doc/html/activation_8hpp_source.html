322  // GAUSSIAN ERROR LINEAR UNITS (GELUS) https://arxiv.org/pdf/1606.08415.pdf
323  // f(x) = 0.5x (1 + tanh[\sqrt{2/}(x + 0.044715x^3)])
324  // df = x ( 1 + tanh[\sqrt{2/}(x + 0.044715x^3)] ) + \sqrt(2/) x sech^2[\sqrt(2/) x (1+0.44715x^2) (1+0.134145x^2) ]
325  // where sec^2(x) = 1 - tanh^2(x)
326  // derivative generated using service from https://www.symbolab.com/solver/derivative-calculator
327  template <Expression Ex>
328  auto inline gelu( Ex const& ex ) noexcept
329  {
330  auto _gelu = []<typename T>( T x )
331  {
332  auto const ans = 0.5 * x * ( 1.0 + std::tanh( 0.79788456080286535588 * x ( 1.0 + 0.044715*x*x ) ) );
333  return static_cast<T>( ans );
334  };
335  auto sech_2 = []( auto x )
336  {
337  return 1.0 - std::pow( std::tanh(x), 2 );
338  };

341  auto const sq_2_pi_x = 0.79788456080286535588 * x;
342  auto const _xx = x * x;
343  auto const ans = 0.5 * ( 1.0 + std::tanh( sq_2_pi_x * ( 1.0 + 0.044715 * _xx ) ) ) + sq_2_pi_x * sech_2( sq_2_pi_x * (1.0 + 0.044715 * _xx ) * ( 1.0 + 0.134145 * _xx) );
