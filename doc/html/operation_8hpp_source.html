650  auto inline img2col( unsigned long const row_kernel, unsigned long col_kernel=-1,
651  unsigned long const row_padding=0, unsigned long col_padding=0,
652  unsigned long const row_stride=1, unsigned long const col_stride=1,
653  unsigned long const row_dilation=1, unsigned long const col_dilation=1 ) noexcept
654  {
655  if ( col_kernel == (unsigned long)-1 ) col_kernel = row_kernel;
656 
657  std::shared_ptr<std::vector<std::uint32_t>> s_index_record = std::make_shared<std::vector<std::uint32_t>>(); // col_img[idx] = img[index_record[idx]] -- (-1) for zero padding
658 
659  auto img2col_forward = [s_index_record]<Tensor Tsor>
660  (
661  Tsor const& input_img, Tsor& output_col_mat,
662  unsigned long kernel_row, unsigned long kernel_col,
663  unsigned long padding_row, unsigned long padding_col,
664  unsigned long stride_row, unsigned long stride_col,
665  unsigned long dilation_row, unsigned long dilation_col
666  ) noexcept
667  {
668  typedef typename Tsor::value_type value_type;
669  std::vector<std::uint32_t>& index_record = *s_index_record; //32 bit should be enough for memory address offeset
670 
671  std::vector<unsigned long> input_shape = input_img.shape();
672  better_assert( input_shape.size() == 4, "Expecting a 4D tensor." );
673  auto const [BS, R, C, CH] = std::make_tuple( input_shape[0], input_shape[1], input_shape[2], input_shape[3] );
674 
675  unsigned long const output_row = ( R + 2 * padding_row - ( dilation_row * (kernel_row - 1) + 1 ) ) / stride_row + 1;
676  unsigned long const output_col = ( C + 2 * padding_col - ( dilation_col * (kernel_col - 1) + 1 ) ) / stride_col + 1;
677  unsigned long const output_column_matrix_row = kernel_row * kernel_col * CH;
678  unsigned long const output_column_matrix_col = BS * output_row * output_col;
679 
680  output_col_mat.resize( {output_column_matrix_row, output_column_matrix_col} );
681 
682  if ( index_record.size() != output_column_matrix_row * output_column_matrix_col ) // first-run?
683  {
684  index_record.resize( output_column_matrix_row * output_column_matrix_col );
685 
686  for ( auto bs : range( BS ) )
687  {
688  std::int64_t const col_offset = bs * output_row * output_col * kernel_row * kernel_col * CH;
689  std::int64_t const im_offset = bs * R * C * CH;
690  for ( auto c : range( CH * kernel_row * kernel_col ) )
691  {
692  std::int64_t const w_offset = c % kernel_col;
693  std::int64_t const h_offset = ( c / kernel_col ) % kernel_row;
694  std::int64_t const c_im = c / ( kernel_col * kernel_row );
695 
696  for ( auto h : range( output_row ) )
697  {
698  std::int64_t const im_row_idx = h * stride_row - padding_row + h_offset * dilation_row;
699  for ( auto w : range( output_col ) )
700  {
701  std::int64_t const im_col_idx = w * stride_col - padding_col + w_offset * dilation_col;
702  std::int64_t const im_idx = im_offset+( im_row_idx * C + im_col_idx ) * CH + c_im;
703  std::int64_t const col_idx = col_offset+( c * output_row + h ) * output_col + w;
704  index_record[col_idx] = static_cast<std::uint32_t>((im_row_idx<0 || im_row_idx>=static_cast<std::int64_t>(R) || im_col_idx<0 || im_col_idx>=static_cast<std::int64_t>(C)) ? 0xffffffff : im_idx);
705  }
706  }
707  }
708  }
709  // re-arrange [bs, new_R, new_C] --> [new_R, new_c*bs]
710  {
711  std::vector<std::uint32_t> re_arranged_index;
712  re_arranged_index.resize( index_record.size() );
713 
714  view_3d<std::uint32_t> re_arranged_mat{ re_arranged_index.data(), output_column_matrix_row, BS, output_row*output_col };
715  view_3d<std::uint32_t> index_record_mat{ index_record.data(), BS, output_column_matrix_row, output_row*output_col };
716 
717  for ( auto bs : range( BS ) )
718  for ( auto r : range( output_column_matrix_row ) )
719  for ( auto c : range( output_row*output_col ) )
720  re_arranged_mat[r][bs][c] = index_record_mat[bs][r][c];
721  // overwrite index record
722  std::copy( re_arranged_index.begin(), re_arranged_index.end(), index_record.begin() );
723  }
724  }
725 
726  // fill-in
727  for ( auto idx : range( output_col_mat.size() ) )
728  {
729  auto const index = index_record[idx];
730  output_col_mat[idx] = (index == 0xffffffff) ? value_type{0} : input_img[index];
731  }
732  };

735  {
736  typedef typename Tsor::value_type value_type;
737  ans.resize( input.shape() );

742  {
743  auto const index = index_record[idx];
744  if ( index != 0xffffffff )
745  ans[index] += grad[idx];
746  }
747  };

750  std::shared_ptr<std::any> back_grad_cache = std::make_shared<std::any>();

772  auto inline conv2d
773  (
774  unsigned long row_input, unsigned long col_input,
775  unsigned long const row_stride=1, unsigned long const col_stride=1,
776  unsigned long const row_dilation=1, unsigned long const col_dilation=1,
777  std::string const& padding="valid"
778  ) noexcept
779  {
780  // lhs_ex is for one 4D tensor of [BS, R, C, CH]
781  // rhs_ex is for NC 4D filter of [1, r, c, CH], thus the shape is [NC, r, c, CH]
782  // the output tensor is of shape [BS, .., .., NC]
783  //
784  // Note: the rhs expression is fixed as a variable, as we need to extract the kernel shape from it
785  //
786  return [row_input, col_input, row_stride, col_stride, row_dilation, col_dilation, padding ]<Expression Ex, Variable Va>( Ex const& lhs_ex, Va const& rhs_ex ) noexcept
787  {
788  std::vector<unsigned long> const& shape = rhs_ex.shape();
789  better_assert( shape.size() == 4 );
790  auto const[new_channel, row_kernel, col_kernel, channel] = std::make_tuple( shape[0], shape[1], shape[2], shape[3] );
791  //TODO: optimization in case of small kernels of (1, 1), (3, 3)
792  unsigned long row_padding = 0;
793  unsigned long col_padding = 0;
794  if ( padding == "same" )
795  {
796  unsigned long const row_padding_total = (row_kernel + (row_kernel - 1) * (row_dilation - 1) - row_stride);
797  better_assert( !(row_padding_total & 0x1), "Expecting total row padding to be even, but got ", row_padding_total );
798  unsigned long const col_padding_total = (col_kernel + (col_kernel - 1) * (col_dilation - 1) - col_stride);
799  better_assert( !(col_padding_total & 0x1), "Expecting total col padding to be even, but got ", col_padding_total );
800  row_padding = ((row_kernel&1)+row_padding_total) >> 1;
801  col_padding = ((col_kernel&1)+col_padding_total) >> 1;
802  }
803 
804  unsigned long const row_output = ( row_input + 2 * row_padding - ( row_dilation * (row_kernel - 1) + 1 ) ) / row_stride + 1;
805  unsigned long const col_output = ( col_input + 2 * row_padding - ( col_dilation * (col_kernel - 1) + 1 ) ) / col_stride + 1;
806 
807  auto lhs_ex_as_col = img2col(row_kernel, col_kernel, row_padding, col_padding, row_stride, col_stride, row_dilation, col_dilation)( lhs_ex ); // [BS, R, C, CH] ==> [r*c*CH, BS*new_row*new_col]
808 
809  auto rhs_ex_flatten = reshape({row_kernel*col_kernel*channel,})( rhs_ex ); // [NC, r, c, CH] ==> [NC, r*c*CH]
810 
811  auto flatten_output = rhs_ex_flatten * lhs_ex_as_col; // [NC, BS * new_row * new_col]
812 
813  auto tr_output = transpose( flatten_output ); // [BS*new_row*new_col, NC]
814 
815  auto ans = reshape({row_output, col_output, new_channel})( tr_output );
816 
817  return ans;
818  };
